## New Example Programs

### 1. Guess the Number
**File:** `src/examples/guess-number.js`

Classic game demonstrating:
- `random(min, max)` for secret number
- `input()` for user guesses
- `num()` for string→number conversion
- Comparison operators and loops

```
// Guess the Number
let secret = random(1, 100)
let guess = 0
let attempts = 0

print "I'm thinking of a number between 1 and 100"

while (guess != secret) {
  print "Your guess:"
  guess = num(input())
  attempts = attempts + 1

  if (guess < secret) {
    print "Too low!"
  } else {
    if (guess > secret) {
      print "Too high!"
    }
  }
}

print "Correct! You got it in " + attempts + " attempts"
```

### 2. Number Guessing (Computer Guesses)
**File:** `src/examples/computer-guesses.js`

Reversed version where computer guesses user's number:
- Binary search algorithm
- `key()` for simple responses (h/l/c)
- Algorithm demonstration

```
// Computer Guesses Your Number
print "Think of a number between 1 and 100"
print "Press 'h' if my guess is too high"
print "Press 'l' if my guess is too low"
print "Press 'c' if I got it correct"
print ""
print "Press any key when ready..."
key()

let low = 1
let high = 100
let guesses = 0

while (low <= high) {
  let mid = (low + high) / 2
  // Round down to integer
  mid = mid - (mid % 1)
  guesses = guesses + 1

  print "Is it " + mid + "? (h/l/c)"
  let response = key()

  if (response == "c") {
    print "I got it in " + guesses + " guesses!"
    low = high + 1
  } else {
    if (response == "h") {
      high = mid - 1
    } else {
      low = mid + 1
    }
  }
}
```

### 3. Simple Calculator
**File:** `src/examples/calculator.js`

Basic arithmetic calculator:
- `input()` for numbers
- `num()` for conversion
- `key()` for operation selection

```
// Simple Calculator
print "Enter first number:"
let a = num(input())

print "Operation (+, -, *, /):"
let op = key()

print "Enter second number:"
let b = num(input())

let result = 0
if (op == "+") {
  result = a + b
} else {
  if (op == "-") {
    result = a - b
  } else {
    if (op == "*") {
      result = a * b
    } else {
      if (op == "/") {
        result = a / b
      }
    }
  }
}

print a + " " + op + " " + b + " = " + result
```

### 4. Dice Roller
**File:** `src/examples/dice.js`

Simple dice simulation:
- `random()` for dice rolls
- `key()` to roll again

```
// Dice Roller
print "Press any key to roll the dice..."
print "(Press 'q' to quit)"
print ""

let total = 0
let rolls = 0
let k = ""

while (k != "q") {
  k = key()
  if (k != "q") {
    let die1 = random(1, 6)
    let die2 = random(1, 6)
    rolls = rolls + 1
    total = total + die1 + die2
    print "Roll " + rolls + ": " + die1 + " + " + die2 + " = " + (die1 + die2)
  }
}

print ""
print "Total rolls: " + rolls
print "Sum of all rolls: " + total
```

### 5. Rock Paper Scissors
**File:** `src/examples/rps.js`

Classic game against computer:
- `random()` for computer choice
- `key()` for player choice

```
// Rock Paper Scissors
print "Rock Paper Scissors!"
print "Press r=rock, p=paper, s=scissors, q=quit"
print ""

let wins = 0
let losses = 0
let ties = 0

let playing = true
while (playing) {
  print "Your choice:"
  let player = key()

  if (player == "q") {
    playing = false
  } else {
    let comp = random(1, 3)
    let compChoice = "rock"
    if (comp == 2) {
      compChoice = "paper"
    } else {
      if (comp == 3) {
        compChoice = "scissors"
      }
    }
    print "Computer: " + compChoice

    // Determine winner
    if (player == "r" and compChoice == "scissors") {
      print "You win!"
      wins = wins + 1
    } else {
      if (player == "p" and compChoice == "rock") {
        print "You win!"
        wins = wins + 1
      } else {
        if (player == "s" and compChoice == "paper") {
          print "You win!"
          wins = wins + 1
        } else {
          if (player == "r" and compChoice == "rock") {
            print "Tie!"
            ties = ties + 1
          } else {
            if (player == "p" and compChoice == "paper") {
              print "Tie!"
              ties = ties + 1
            } else {
              if (player == "s" and compChoice == "scissors") {
                print "Tie!"
                ties = ties + 1
              } else {
                print "You lose!"
                losses = losses + 1
              }
            }
          }
        }
      }
    }
    print ""
  }
}

print "Final: " + wins + " wins, " + losses + " losses, " + ties + " ties"
```
